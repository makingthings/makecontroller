00068 /** \defgroup Io
00069   A mechanism to manage the 32 parallel io lines on the controller.
00070 
00071   Control of the SAM7X's IO lines is more complex than most microcontrollers.  All 
00072   lines may be inputs or outputs, have pull-ups or not, have a glitch filter enabled 
00073   or not, and so on.  In addition all IO pins serve at least double and sometimes 
00074   triple duty, being general IO lines and also being IO lines for one or more of 
00075   the controller's many on-chip peripherals.  All these functions can be controlled
00076   from the Io subsystem.
00077 
00078   This API can be used in two ways - either one line at a time or many at a time.  
00079   Functions supporting the latter are suffixed with "Bits".  So the one-at-a-time 
00080   mechanism for setting an IO to true is called Io_SetTrue().  It takes an index and
00081   the ones for setting
00082   a number of lines is Io_SetTrueBits( ).
00083 
00084   The pattern of use is as follows:  prior to using an IO line, Call Io_Start() on it, the 
00085   return value indicates if the start operation was successful.  After this, call any 
00086   of the other Io functions.
00087 
00088   \todo Consider changing the api slightly to permit the immediate use of the io
00089         without starting it.
00090   \todo Add glitch filter control
00091   \todo Implement more bits-style functions
00092   \ingroup Controller
00093   @{
00094 */

00096 int Io_Start( int index, bool lock )
00097 {
00098   if ( index < 0 || index > IO_PIN_COUNT )
00099     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00100   IoPin* p = &Io.pins[ index ];
00101   // If it's not locked at all and 
00102   //    either no lock is requested or no a lock is requested but no other are using the io and
00103   //    there are not a huge number of users
00104   if ( !p->lock && ( !lock || ( lock && p->users == 0 ) ) && ( p->users < IO_USERS_MAX - 2 ) )
00105   {
00106     if ( Io.users == 0 )
00107       Io_Init();
00108    
00109     // Lock it up.
00110     Io.users++;
00111     p->users++;
00112     if ( lock )
00113       p->lock = true;
00114 
00115     // Output Pair 1
00116     // Switch to PIO
00117     // AT91C_BASE_PIOA->PIO_PER = AT91C_PIO_PA26;
00118     // AT91C_BASE_PIOB->PIO_PER = AT91C_PIO_PB23;
00119     
00120     return CONTROLLER_OK;
00121   }
00122   else
00123     return CONTROLLER_ERROR_TOO_MANY_USERS;
00124 }

00146 int Io_StartBits( longlong bits, bool lock )
00147 {
00148   IoPin* p = &Io.pins[ 0 ];
00149   int i = 0;
00150   int failure = -1;
00151   ullong mask = 1LL;
00152   for ( i = 0; i < 64; i++ )
00153   {
00154     if ( bits & mask )
00155     {
00156       if ( !p->lock && ( !lock || ( lock && p->users == 0 ) ) && p->users < IO_USERS_MAX - 2 )
00157       {
00158         if ( Io.users == 0 )
00159           Io_Init();
00160         if ( lock )
00161           p->lock = true;
00162         p->users++;
00163         Io.users++;
00164       }
00165       else
00166       {
00167         failure = i;
00168         break;
00169       }
00170     }
00171     mask <<= 1;
00172     p++;
00173   }
00174 
00175   if ( failure != -1 )
00176   {
00177     IoPin* p = &Io.pins[ 0 ];
00178     mask = 1LL;
00179     for ( i = 0; i < failure; i++ )
00180     {
00181       if ( bits & mask )
00182       {
00183         if ( lock )
00184           p->lock = false;
00185         p->users--;
00186         Io.users--;
00187         if ( Io.users == 0 )
00188           Io_Deinit();
00189       }
00190       mask <<= 1;
00191       p++;
00192     }
00193     return CONTROLLER_ERROR_CANT_LOCK;
00194   }
