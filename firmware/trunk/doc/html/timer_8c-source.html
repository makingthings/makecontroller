00054 /**

00057   @return Zero on success.

00060 int Timer_SetActive( bool active )
00061 {
00062   if ( active )
00063   {
00064     if ( Timer.users++ == 0 )
00065     {
00066       int status;
00067   
00068       status = Timer_Init();

00070       {
00071         Timer.users--;

00073       }
00074     }
00075   }
00076   else
00077   {
00078     if ( --Timer.users == 0 )

00121 /** Sets the requested entry to run.
00122   This routine adds the entry to the running queue and then decides if it needs
00123   to start the timer (if it's not running) or alter the timer's clock for a shorter
00124   period.
00125   @param timerEntry pointer to the FastTimerEntry to be run. 
00126   */
00127 int Timer_Set( TimerEntry* timerEntry )
00128 {
00129   // this could be a lot smarter - for example, modifying the current period?
00130   if ( !Timer.servicing ) 
00131     TaskEnterCritical();
00132 
00133   if ( !Timer.running )
00134   {
00135     Timer_SetActive( true );
00136     Timer_SetTimeTarget( timerEntry->timeInitial );
00137     Timer_Enable();
00138   }  
00139 
00140   // Calculate how long remaining
00141   int target = Timer_GetTimeTarget();
00142   int timeCurrent = Timer_GetTime();
00143   int remaining = target - timeCurrent;
00144 
00145   // Get the entry ready to roll
00146   timerEntry->timeCurrent = timerEntry->timeInitial;
00147 
00148   // Add entry
00149   TimerEntry* first = Timer.first;
00150   Timer.first = timerEntry;
00151   timerEntry->next = first;
00152 
00153   // Are we actually servicing an interupt right now?
00154   if ( !Timer.servicing )
00155   {
00156     // No - so does the time requested by this new timer make the time need to come earlier?
00157     if ( timerEntry->timeCurrent < ( remaining - TIMER_MARGIN ) )
00158     {
00159       // Damn it!  Reschedule the next callback
00160       Timer_SetTimeTarget( target - ( remaining - timerEntry->timeCurrent ));
00161     }
00162     else
00163     {
00164       // pretend that the existing time has been with us for the whole slice so that when the 
00165       // IRQ happens it credits the correct (reduced) time.
00166       timerEntry->timeCurrent += timeCurrent;
00167     }
00168   }
00169   else
00170   {
00171     // Yep... we're servicing something right now
00172 
00173     // Make sure the previous pointer is OK.  This comes up if we were servicing the first item
00174     // and it subsequently wants to delete itself, it would need to alter the next pointer of the 
00175     // the new head... err... kind of a pain, this
00176     if ( Timer.previous == NULL )
00177       Timer.previous = timerEntry;
00178 
00179     // Need to make sure that if this new time is the lowest yet, that the IRQ routine 
00180     // knows that.  Since we added this entry onto the beginning of the list, the IRQ
00181     // won't look at it again
00182     if ( Timer.nextTime == -1 || Timer.nextTime > timerEntry->timeCurrent )
00183         Timer.nextTime = timerEntry->timeCurrent;
00184   }
00185 
00186   if ( !Timer.servicing ) 
00187     TaskExitCritical();
00188 
00189   return CONTROLLER_OK;
00190 }

00192 /** 
00193   Cancel a timer event.
00194   @param timerEntry The entry to be removed.
00195   @return 0 on success.
00196   @see Timer_Start, Timer_Set and Timer_Start
00197 */
00198 int Timer_Cancel( TimerEntry* timerEntry )
00199 {
00200   if ( !Timer.servicing ) 
00201     TaskEnterCritical();
00202 
00203   // Look through the running list - clobber the entry
00204   TimerEntry* te = Timer.first;
00205   TimerEntry* previousEntry = NULL;
00206   while ( te != NULL )
00207   {
00208     // check for the requested entry
00209     if ( te == timerEntry )
00210     {
00211       // remove the entry from the list
00212       if ( te == Timer.first )
00213         Timer.first = te->next;
00214       else
00215         previousEntry->next = te->next;
00216       
00217       // make sure the in-IRQ pointers are all OK
00218       if ( Timer.servicing )
00219       {
00220         if ( Timer.previous == timerEntry )
00221           Timer.previous = previousEntry;
00222         if ( Timer.next == timerEntry )
00223           Timer.next = te->next;
00224       }
00225 
00226       // update the pointers - leave previous where it is
00227       te = te->next;
00228     }
00229     else
00230     {
00231       previousEntry = te;
00232       te = te->next;
00233     }
00234   }
00235 
00236   if ( !Timer.servicing ) 
00237     TaskExitCritical();
00238 
00239   return CONTROLLER_OK;
00240 }

00284 int Timer_Init()
00285 {
00286   Timer.first = NULL;
00287 
00288   Timer.count = 0;
00289   Timer.jitterTotal = 0;
00290   Timer.jitterMax = 0;  
00291   Timer.jitterMaxAllDay = 0;
00292 
00293   Timer.running = false;
00294   Timer.servicing = false;
00295 
00296   // Configure TC by enabling PWM clock
00297   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_TC0;
00298                                     
00299   unsigned int mask ;
00300   mask = 0x1 << AT91C_ID_TC0;
00301    
00302   /* Disable the interrupt on the interrupt controller */
00303   AT91C_BASE_AIC->AIC_IDCR = mask;
00304 
00305   /* Save the interrupt handler routine pointer */  
00306   AT91C_BASE_AIC->AIC_SVR[ AT91C_ID_TC0 ] = (unsigned int)Timer_Isr;
00307 
00308   /* Store the Source Mode Register */
00309   // 4 PRIORITY is random
00310   AT91C_BASE_AIC->AIC_SMR[ AT91C_ID_TC0 ] = AT91C_AIC_SRCTYPE_INT_HIGH_LEVEL | 4  ;
00311   /* Clear the interrupt on the interrupt controller */
00312   AT91C_BASE_AIC->AIC_ICCR = mask ;
00313 
00314   // Set the timer up.  We want just the basics, except when the timer compares 
00315   // with RC, retrigger
00316   //
00317   // MCK is 47923200
00318   // DIV1: A tick MCK/2 times a second
00319   // This makes every tick every 41.73344ns
00320   // DIV2: A tick MCK/8 times a second
00321   // This makes every tick every 167ns
00322   // DIV3: A tick MCK/32 times a second
00323   // This makes every tick every 668ns  
00324   // DIV4: A tick MCK/128 times a second
00325   // This makes every tick every 2.671us
00326   // DIV5: A tick MCK/1024 times a second
00327   // This makes every tick every 21.368us
00328   // CPCTRG makes the RC event reset the counter and trigger it to restart
00329   AT91C_BASE_TC0->TC_CMR = AT91C_TC_CLKS_TIMER_DIV5_CLOCK | AT91C_TC_CPCTRG;
00330                    
00331   // Only really interested in interrupts when the RC happens
00332   AT91C_BASE_TC0->TC_IDR = 0xFF; 
00333   AT91C_BASE_TC0->TC_IER = AT91C_TC_CPCS; 
00334 
00335   // load the RC value with something
00336   AT91C_BASE_TC0->TC_RC = 0xFFFF;
00337 
00338   // Enable the interrupt
00339   AT91C_BASE_AIC->AIC_IECR = mask;
00340 
00341   return CONTROLLER_OK;
00342 }
